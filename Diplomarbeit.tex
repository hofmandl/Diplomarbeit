\chapter{Einleitung}

Sei $ (\Omega,F,P) $ ein Wahrscheinlichkeitsraum. Sei $W$ eine auf diesen Raum definierte Brownsche
Bewegung. Wir betrachten das Black-Scholes-Marktmodell mit einer Aktie und konstanten Koeffizienten, die
Preisprozesse seien also durch die folgenden Gleichungen gegeben:
\begin{align*}
 dB(t) &= r B(t) dt \\
  B(0) &= 1         \\
 dS(t) &= S(t)(\mu dt + \sigma dW(t)) \\
  S(0) &= s_0
\end{align*}

Sei $X$ der Vermögensprozess zur Handelsstrategie $\pi$. Er ist durch folgende Differentialgleichung gegeben:
\begin{align*}
 dX(t) &= X(t)(r dt + \pi(t)(\mu -r) dt + \sigma dW(t)) \\
  X(0) &= x_0
\end{align*}
Sei $U$ eine Nutzenfunktion. Dann nennen wir
\begin{equation*}
  \underset{\pi}{\text{max}} E(U(X^{\pi}(T)))
\end{equation*}
das \emph{klassische Portfoliooptimierungsproblem}. Wir diskutieren seine Lösung mittels stochastischer
Steuerung, welche auf Merton zurüchgeht.

Das Black-Scholes-Modell wurde vielfach dafür kritisiert, dass seine Preisprozesse stetig sind. In Realität
beobachtet man aber Crashes von Aktienmärkten. Um dies zu modellieren machen wir nun einen anderen Ansatz als
den vielfach diskutierten Vorschlag den Aktienprozess $S$ durch einen unstetigen Prozess zu modellieren: Wir
betrachten eine Stoppzeit $\tau$, zu der die Aktie um den Faktor $k$ fällt. Das Endvermögen hat in
diesem Modell dann die folgende Form:
\begin{equation*}
  X^{\pi \tau}(T)=(1-\pi(\tau)k)X(T)
\end{equation*}

Nun betrachten wir den Markt als Gegenspieler des Investors und versuchen das folgende Maximierungsproblem
\begin{equation*}
  \underset{\pi}{\text{max}} \underset{\tau}{\text{min}} E(U(X^{\pi\tau}(T)))
\end{equation*}
zu lösen, das wir als \emph{Worst-Case Portfoliooptimierungsproblem} bezeichnen. Wir diskutieren
zwei von Ralf Korn vorgeschlagene Lösungen für dieses Problem. Die erste Lösung bestimmt die
Handelstrategie durch Gleichgewichtsüberlegungen. Die zweite Lösung macht einen ganz anderen
Ansatz, durch Aufstellung eines Systems von HJB-Gleichungen kann eine Wertfunktion ermittelt
werden, aus der dann auf die optimale Handelstrategie geschlossen werden kann. Dabei werden wir
insbesondere untersuchen, inwiefern sich diese beiden Ansätze in ihren Ergebnissen,in ihrer
Beweismethodik und ihren weiteren Anwendungsmöglichkeiten unterscheiden.

Um konkret rechen zu können, werden wir sowohl das \emph{klassische Portfoliooptimierungsproblem} als auch
die beiden Ansätze beim \emph{Worst-Case Portfoliooptimierungsproblem} nach Wahl der logarithmischen
Nutzenfunktion diskutieren. 

Schliesslich soll noch untersucht werden, inwieweit wir das Problem als ein optimales Stoppproblem, welches
der Markt als Gegenspieler des Investors zu lösen hat, formuliert werden kann.

\section{Finanzmarktmodell}
\begin{myDefinition} \label{Standardfinanzmarktmodell}
Sei $(\Omega,F,P) $ ein Wahrscheinlichkeitsraum. Sei $W$ eine auf diesen Raum definierte Brownsche
Bewegung. Seien $r$, $\mu$, $\sigma$ reelle Zahlen und sei $r < \mu$ und $\sigma>0$. Der Bond $B$ und die
Aktie $S$ seien die eindeutigen Lösungen der stochastischen Differentialgleichungen
\begin{align}
 dB(t) &= r B(t) dt \\
  B(0) &= 1         \\
 dS(t) &= S(t)(\mu dt + \sigma dW(t)) \\
  S(0) &= s_0
\end{align}
Gelten alle diese Vorrausetzungen, so sagen wir dass die Vorrausetzungen unseres Standardfinanzmarktmodells
gelten.
\end{myDefinition}

\begin{myDefinition} \label{Handelsstrategie}
Eine Handelsstrategie $\pi$ ist ein stochastischer Prozess auf dem Wahrscheinlichkeitsraum des
Standardfinanzmarktmodells.
\end{myDefinition}

\begin{myDefinition} \label{Vermögensgleichung}
Es gelten die Vorrausetzungen unseres Standardfinanzmarktmodells. Die Lösung von
\begin{align} \label{eq:Vermögensgleichung}
 dX(t) &= X(t)(r dt + \pi(t)(\mu -r) dt + \sigma dW(t)) \\
  X(t_0) &= x_0
\end{align}
bezeichen wir mit $X_{\pi,t_0,x}$. Ist $t_0=0$, so schreiben wir auch $X_{\pi,x}$. 
\end{myDefinition}

\begin{myDefinition} \label{def:f}
Es gelten die Vorrausetzungen unseres Standardfinanzmarktmodells. Sei $f$ durch 
\begin{equation} \label{eq:fvonpi}
 f(\pi) =  \pi (\mu-r)-\einhalb \pi^2 \sigma^2
\end{equation}
definiert.
\end{myDefinition}

\begin{mySatz} \label{Eigenschaften_f}
$f$ besitzt die beiden Nullstellen
\begin{align}

 x_1 &= 2 \frac{\pi-\mu}{\sigma^2},
\end{align}
nimmt ihr Maximum bei
\begin{equation} \label{eq:pimax}
 \pi_{max} =  \pistern
\end{equation}
an hat dort den Wert $\einhalb \frac{(\pi-\mu)^2}{\sigma^2}$.
\end{mySatz}

\section{Anwendung auf das Portfolioproblem}
\begin{myDefinition}
 Sei $U:(0,\infty) \rightarrow R$ strikt konkav, stetig differenzierbar und es gelte
\begin{equation}
 \lim U'(x)=\infty, \qquad \lim U'(x)=0.
\end{equation}
Dann nennen wir $U$ eine Nutzenfunktion.
\end{myDefinition}



\begin{myDefinition}
\standardsatz
Wir bezeichen mit
\begin{equation}
 J(x,t,\pi,U) = E(U(X_{\pi,x,t}(T))
\end{equation}
den erwarteten Endnutzen durch Verfolgung einer Handestrategie $\pi$ bezüglich $U$.
Die Funktion
\begin{equation}
 v_0^U(t,x) = \sup J(x,t,\pi,U)
\end{equation}
heißt Wertfunktion.
\end{myDefinition}
Die $0$ im subscript der Wertefunktion wird erst in späteren Kapiteln klar werden. 

\begin{mySatz}
Ist $V$ eine Lösung von 
\begin{equation} \label{eq:HJBPortfolio} 
 \underset{\pi \in (a,b)}{\text{max}} ( \einhalb \pi^2 \sigma^2 x^2 V_{xx}(t,x) + (r+\pi (b-r) x)
V_{x}(t,x) + V_t(t,x)=0  )
\end{equation}
mit
\begin{equation}
 V(T,x)=U(x)
\end{equation}
Dann gilt 
\begin{equation}
 V(t,x)=v_0^U(t,x)
\end{equation}
und
\begin{equation}
 p(t) 
\end{equation}
ist der optimale portfolioprozess.

Ist $\pi$ argmin, so ist $\pi$ die zugehörige Handelsstrategie.


\end{mySatz}


\section{Explizite Lösung für logaritmischen Nutzen}
Wir zeigen, dass die Wertfunktion durch
\begin{equation} \label{eq:LogWertfunktion}
 v_0(t,x)= \log(x)+\left(r + \einhalb \bruch^2 \right) (T-t)
\end{equation}
ist. Dazu berechen wir die Ableitungen
\begin{equation}
 V_t(t,x)= -(r+\frac{1}{2} (\frac{\mu-r}{\sigma})^2,
\end{equation}
\begin{equation}
 V_x(t,x)= \frac{1}{x}
\end{equation}
und
\begin{equation}
 V_x(t,x)= -\frac{1}{x^2}.
\end{equation}
Einsetzen in Korn/Korn, Seite 276 ergibt:
\begin{align*}
 B &= \einhalb \pi^2 \sigma^2 x^2 V_{xx}(t,x) + (r+\pi(b-r)) x V_x(t,x) + V_t(t,x) \\
   &= \einhalb \pi^2 \sigma^2 x^2 \frac{1}{ \cancel{x^2}} + (r+\pi(b-r)) x \frac{1}{x} + -r+\frac{1}{2}
         (\frac{\mu-r}{\sigma})^2\\
  &= -\pi^2\einhalb \sigma^2+\pi(b-r)-\einhalb (\frac{\mu-r}{\sigma})^2 \\
  &= f(\pi) -\einhalb (\frac{\mu-r}{\sigma})^2
\end{align*}
Dabei haben wir in der letzten Gleichung \ref{eq:fvonpi} eingesetzt. Das Maximum von $f$ ist $\einhalb
(\frac{\mu-r}{\sigma})^2$, Maximieren der vorhergehenden Gleichung ergibt also tatsächlich $0$. Ausserdem
erfüllt \ref{eq:LogWertfunktion} auch die Endbedingung: 
\begin{equation}
 V(T,x)= (r+\frac{1}{2} (\frac{\mu-r}{\sigma})^2 (T-T)= \log(x)+0 = \log(x)
\end{equation}

Maximiert wird die Gleichung durch
\begin{equation}
 \pi=\pistern,
\end{equation}
das ist dann der optimale Portfolioprozess.


Zur Probe berechen wir auch noch den besten konstanten Portfoliprozess, dabei erwarten wir natürlich das
gleiche Ergenis. Wir müssen dazu nur das $\pi$ finden, welches \ref{eq:ExpLogUtilityKonst} maximiert. Dazu
setzen wir \ref{eq:fvonpi} in \ref{eq:ExpLogUtilityKonst} ein 

\begin{equation}
 \pi=\pistern,
\end{equation}

\begin{align}
E(log(X(t))) & = log(x)  + (r + \pi(\mu -r)-\frac{1}{2}\pi^2 \sigma^2) t \\
              &= log(x)  +  (r+f(\pi))t 
\end{align}
Und man erkennt das wir das Maximum von $f$ suchen. Das haben wir schon in \ref{eq:pimax} als
\begin{equation}
 \pi=\pistern
\end{equation}
und wir sind in unserer Probe bestätigt worden.


\chapter{Ein spieltheoretischer MaxMin-Ansatz zur Lösung des Portfolioproblems}
Das Modell unseres Standardfinanzmarktes hat den Nachteil, dass es den Aktienekurs als stetige Funktion der
Zeit modelliert. In der Realität beobachtet man aber Kurssprünge, insbesondere nach unten, die man als Crash
bezeichnet.

Das modellieren wir nun dadurch, dass wir annehmen dass innerhalb des Zeithorizonts es zu einem
Fall des Aktienkuses um das k-fachen seines Wertes kommen kann. Dabei gelte 
$k\in (0,\kstar) $ und $0<\kstar<1$. Bezeichnet $t$ den Crashzeitpunkt, so lässt sich dieser Sachverhalt
mittels der folgenden Formel  ausgedrücken
\begin{equation*}
  k=\frac{X_{t-}-X_t}{X_{t-}} \Leftrightarrow X_t=(1-k)X_{t-}.
\end{equation*}
Auf welchen Faktor schrumpft das Vermögen bei einem Crash? Anders ausgedrückt, welches $a$ löst die Gleichung
\begin{equation*}
  a X_{t-}=X_t?
\end{equation*}
Dazu führen wir folgende Rechnung durch
\begin{align}
 a X_{t-} &= \underbrace{(1-\pi(t)) X_{t-}}_{\text{Bondvermögen}}  + \underbrace {\pi(t) (1-k)
X_{t-}}_{\text{gecrashtes Aktienvermögen}} \\
          &= (1-\pi(t)k) X_{t-}, 
\end{align}
also $a=1-\pi(t)k$. Wie sieht nun das zu einem Vermögensprozess gehörige Endvermögen aus, wenn man in unserem
Chrashscenario handelt.Sei $f(x,t_1,t_2)$ das Endvermögen das Endvemögen, wenn man beginnend mit dem Vermögen
$x$ in $t_1$ bis zum Zeitpunkt $t_2$ gemäß dem Vermögensprozess $\pi$ handelt. Den Wert von $f$ haben wir in
\ref{eq:LösVermögensgleichung} auch explizit angegeben. Für das endvermögen ist dann folgender Ansatz sinnvoll
\begin{equation*}
  X(T)= f((1-\pi(t)k) f(x,0,t),t,T),
\end{equation*}
denn bis zum Crashzeitpunkt $t$ sammelt man das Vermögen $f(x,0,t)$ an, der Crash verringert es auf
$(1-\pi(t)k) f(x,0,t)$ und dann hat man noch bis $T$ Zeit es weiter zu vermehren. Mit
\ref{eq:LösVermögensgleichung} berechnen wir nun:
\begin{align}
  X(T) &= (1-\pi(t)k) f(x,0,t) \exp(\int_t^T \dots ) \\
       &= (1-\pi(t)k) x \exp(\int_0^t \dots )  \exp(\int_t^T \dots ) \\
       &= (1-\pi(t)k) x \exp(\int_0^T \dots ) \\
       &= (1-\pi(t)k) f(x,0,T)
\end{align}
Der Crash beinflusst das Endvermögen also nur um einen (zeitabhängigen) Faktor. Das hängt mit der
Exponentialgestalt des Vermögensprozesses zusammen, der relative Vermögenszuwachs hängt nämlich nicht von der
Höhe der Vermögens ab.

Diese mehr oder weniger heuristische Überlegungen motivieren nun das folgende Modell für das Endvermögen.
Startet man zur Zeit $s$ mit dem Anfangsvermügen $x$ und erlebt man zur Zeit $s$ einen Crash der Höhe $k$,
dann hat man in $T$ das Vermögen:
\begin{equation} 
 \xdach^{\pi,t,k}(T) = (1-\pi(t)k) X^{\pi}(T).
\end{equation}


In \cite{Korn2002} wird nun vorgeschlagen eine Strategie $\pi_{opt}$ mit

\begin{equation}\label{eq:kornmaxmin}
  \underset{\pi}{\text{max}} \underset{t,k}{\text{min}} E(U(X^{\pi\tau}(T)))
         = \underset{\tau}{\text{min}} E(U(X^{\pi_{opt}\tau}(T)))
\end{equation}
zu suchen. Dieser Ansatz stammt aus der Spieltheorie, man versucht also den minimalen Nutzen zu maximieren.
Diesen MaxMin-Ansatz möchte ich gundsätzlich übernehmen, kritisiere die Formulierung \ref{eq:kornmaxmin}
aber aus zweierlei Gründen. Zum einen wollen wir auch Strategien modellieren, die auf den Crash
reagieren können. Nach dem Crash wecheln wir in die optimale Stategie des crashfreien settings. Das wird hier
aber nicht mitmodelliert. wir können das ausdrücken, indem wir versuchen
\begin{equation} \label{eq:WCCrashScenario}
   \underset{t,k}{\text{min}} E(v(t,\xdach^{\pi,t,k}(T))
\end{equation}
zu maximieren. Zum anderen soll der Markt auch die Möglichkeit haben, gar nicht zu crashen. In der gegebenen
Formulierung könnte man argumentieren, dass sei durch einen Crash der Höhe $k=0$ mitmodelliert. Wir wollen
das aber explizit machen, indem wir versuchen das Minimum aus \ref{eq:WCCrashScenario} und dem dem erwarteten
Nutzen im Crashfreien setting zu maximieren. Das ist in der Modellierung sauberer und wird auch später die
Beweise durchsichtiger machen. Dann ist es auch nicht mehr nötig, über verschieden Crashhöhen zu minimieren.
Ist unsere Aktienposition positiv, so schadet uns ein Crash der maximalen Größe am meisten, ist sie negativ
(wenn wir also Aktien leerverkauft haben), so schadet uns das Ausbleiben eines Crashes am meisten. 
Alle diese Gedanken berücksichtigend, gelangen wir so zur folgenden Definition:

\begin{myDefinition} \label{def:WCPortfolioproblem}
 Jeder Handelsstrategie ordnen wir die Worst-Case-Schranke
\begin{equation*}
 WC(\pi) =  min (\underset{t}{\text{inf}} \quad E(v(t,X(t)(1-\pi(t)\kstar))),E(U(X(T))))
\end{equation*}
zu. Die Aufgabe
\begin{equation*}
 max \qquad WC(\pi)
\end{equation*}
bezeichnen wir als Worst-Case-Portfoliooptimierungsproblem. Eine Strategie, für die das Maximimum angenommen
wird bezeichen wir als optimale Strategie.
\end{myDefinition}

Wir berechen die Worst-Case-Schranke im Log-utility-modell der reinen Bondstrategie. Intuitiv ist klar, das
der Worst-Case das Ausbleiben eines Crashes ist. Denn ein Crash verursacht uns keinen Schaden, aber gibt uns
die Möglichkeit in die optimale Stategie des crashfreien Settings zu wechseln. Das machen wir nochmal
explizit. Sei also $\pi_0=0$.
\begin{align*}
 WC(\pi_0) &=  \min (\underset{t}{\text{inf}} \quad E(v_0(t,X(t)(1-\pi_0(t)\kstar))),E(\log(X(T)))) \\
           &=  \min (\underset{t}{\text{inf}} \quad v_0(t,x\exp(rt)),\log(x\exp(rt)) \\
           &=  \min (\underset{t}{\text{inf}} \quad \log(x) + rt +\left[ r + \einhalb \bruch^2
\right](T-t),\log(x\exp(rt)) \\
           &=  \min (\underset{t}{\text{inf}} \quad \log(x) +\left[\einhalb \bruch^2
\right](T-t) +rT,\log(x\exp(rt)) \\
          &= \log(x) +rT,
\end{align*}
denn in der letzen Zeile ist klar, dass das $inf$ im ersten Argunement von $\min$ bei $t=T$ anegnommen wird.
Wir halten also
\begin{equation} \label{eq:worstcasereinbond}
 WC(\pi_0) = \log(x) +rT,
\end{equation}
fest.

Als nächstes berechen wir die Worst-Case-Schranke, wenn man der optimalen Startegie $\pistar=\pistern$ des
crashfreien Scenarios verfolgt:
\begin{equation} 
 WC(\pistar) = \log(x) +rT + \einhalb \bruch^2 T + \log(1-\pistar\kstar),
\end{equation}

Der Unterschied zwischen diesen beiden Schranken liegt also im Term $\einhalb \bruch^2 T +
\log(1-\pistar\kstar)$. Nur wenn der Zeithorizont groß genug ist ist, wird dieser Term positiv.Dann überwiegt
die höhere Rendite der Aktie den Schaden der uns einmal durch den Crash zugefügt wurde. Eine optimale
Stategie sollte diese Effekte ausbalancieren und je näher man ssich crashfrei dem Ende des zeithorizonts
genähert hat, desto vorsichtiger sollte man in AKtien investieren.

Wir müssen uns nun noch um die Klasse der Handelsstrategien Gedanken machen. Um eine Pleite zu verhindern
müssen wir die Wert der Handelstrategie auf $<\frac{1}{\kstar}$ einschränken.


\begin{mySatz}
Es gilt: $WC(\pi^{+}) \geq WC(\pi)$
\end{mySatz}

\begin{proof}
 Der Beweis ist so organisiert, dass wir erst seine Struktur zeigen, und dann die Details ausarbeiten. Sei
\begin{align*}
 A &:= \underset{t}{\text{inf}} \quad E(v(t,\xdach^{\pi,t,\kstar}(T))) \\
 B &:= E(U(X^{\pi}(T))) \\
 A^+ &:= \underset{t}{\text{inf}} \quad E(v(t,\xdach^{\pi^+,t,\kstar}(T))) \\
 B^+ &:= E(U(X^{\pi^+}(T))) \\
\end{align*}
Wir haben also
\begin{equation}
 min(A,B) \leq min(A^+,B^+)
\end{equation}
zu zeigen.
Wir zeigen zunächst, dass im crashfreien Scenario $\pi^+$ einen höheren erwarteten Nutzen bringt, also
\begin{equation}
 B\leq B^+
\end{equation}
Wenn $A\leq A^+$ sind wir wegen $min(A,B)\leq B \leq B^+$ und $min(A,B)\leq A \leq A^+$, also auch $min(A,B)
\leq min(A^+,B^+)$ fertig.
Sei also $A>A^+$. Dann gilt
\begin{equation}
 \text{für alle t} \qquad \pi(t) < 0
\end{equation}
und
\begin{equation}
 A^+=B^+.
\end{equation}
In dem Fall ist das Worst-Case-Scenario aber das Ausbleiben eines Crashes, also
\begin{equation}
 A \geq B,
\end{equation}
und dann ist die risikofreie Bondstrategie die beste, also
\begin{equation}
 B \leq B^+,
\end{equation}
Insgesamt folgt also
\begin{equation}
 min(A,B)=B \leq B^+ = A^+ = min(A^+,B^+)
\end{equation}

% \end{equation}
% und das zeigt wegen \ref{eq:ExpLogUtility} die Behauptung.

\end{proof}

\begin{mySatz}

Ist $\pidach$ eine Lösung der Differentialgleichung
\begin{equation} \label{eq:dgl}
\dot{\pi(t)}=\frac{1}{k}(1-\pi(t)\kstar) 
\left( 
\pi(t)(\mu-r) - \einhalb 
  \left(
    \pi(t)^2 \sigma^2 + \bruch^2
  \right)
\right)
\end{equation}
mit 
\begin{equation}
\pi(T)=0
\end{equation}
und 
\begin{equation}
0 \leq \pi < \frac{1}{k*}, 
\end{equation}
dann gilt
\begin{equation} \label{eq:Gleichgewichtsbedingung}
v_0(t,x (1-\pi(t)\kstar))= E(\log(X^{\pidach,t,x}(T)))
\end{equation}
\end{mySatz}

\begin{proof}
Wir formen \ref{eq:dgl} zunächst um:
\begin{equation}
\frac{\dot{\pi(t)} \kstar}{(1-\pi(t)\kstar)} =
\left( 
\pi(t)(\mu-r) - \einhalb 
  \left(
    \pi(t)^2 \sigma^2 + \bruch^2
  \right)
\right)
\end{equation}

Integrieren von $t$ nach $T$ ergibt:
\begin{equation}
\left[
- \log(1-\pi(s)\kstar)  
\right]^T_t
=
\int_t^T
\pi(u)(\mu-r) - \einhalb 
  \left(
    \pi(u)^2 \sigma^2 + \bruch^2
  \right)
du
\end{equation}
Berechnung der linken Seite unter Beachtung von $\pi(T)=0$ und herausziehen des bezüglich $u$ konstanten Teil
des Integrals auf der rechten Seite ergibt
\begin{equation}
\log(1-\pi(t)\kstar)
=
\int_t^T
\pi(u)(\mu-r) - \einhalb 
  \left(
    \pi(u)^2 \sigma^2 
  \right)
du
- \einhalb \bruch^2 (T-t)
\end{equation}
verinfachen.



Addieren wir  $\einhalb \bruch^2 (T-t)$ zu dieser Gleichung, so erhalten wir
\begin{equation} \label{eq:umgeformteDGL}
\log(1-\pi(t)\kstar) +  \einhalb \bruch^2 (T-t)
=
\int_t^T
\pi(u)(\mu-r) - \einhalb 
  \left(
    \pi(u)^2 \sigma^2
  \right)
du
\end{equation}

Damit erhalten wir 
\begin{align}
v_0(t,(1-\pi(t)\kstar)) 
 &= \log(1-\pi(t)\kstar) +  \einhalb \bruch^2 (T-t) \qquad  \text{\ref{eq:LogWertfunktion}} \\
 &= \int_t^T
\pi(u)(\mu-r) - \einhalb 
  \left(
    \pi(u)^2 \sigma^2
  \right)
du
\qquad  \text{\ref{eq:umgeformteDGL}} \\
&= E(\log(X'^{\pi})) \qquad  \text{to be refenrencde}
\end{align}
\end{proof}

\begin{mySatz}
$\pidach < \pistar$
\end{mySatz}

\begin{mySatz}
Die Lösung der Differentalgleichung \ref{eq:dgl} ist eine Lösung des Worst-Case-Problems.
\end{mySatz}

\begin{proof}
Die Struktur dieses Beweises ist wie folgt. Wie betrachten in Fallunterscheidungen verschiedene
Handelsstrategien die sich bezüglich $\pidach$ wie folgt unterscheiden.
\begin{itemize}
 \item Handelstrategien mit einer höherem erwarteten Nutzen im crashfreien Scenario ...
  \begin{itemize}
   \item ...und einem höheren Aktienanteil zu Beginn.
   \item ... und einem niedrigerem Aktienanteil zu Beginn.
   \end{itemize}
 \item Handelsstrategien mit einem niedrigeren erwarteten Nutzen im im crashfreien Scenario.
\end{itemize}
Für jeden Fall zeigen wir dann das $\pidach$ jeweils eine bessere Worstcaseschranke besitzt. Damit muss
$\pidach$ dann die optimale Stategie sein.

Sei also $\pi$ eine Handesstrategie  mit einer höheren höherem erwarteten Nutzen im crashfreien Scenario. Aus
\ref{eq:ExpLogUtility} folgt dann
\begin{equation}
 E(\int_{0}^{T} f(\pi(t)) dt) \geq \int_{0}^{T}f(\pidach(t))dt.
\end{equation}
Weiter gilt
\begin{align}
      \int_{0}^{T} f(E(\pi(t))) dt)
 \geq \int_{0}^T E(\pi(t)) (\mu -r) - \einhalb E(\pi(t))^2 \sigma^2 dt \\
 \geq \int_{0}^T E(\pi(t)) (\mu -r) - \einhalb E(\pi(t)^2) \sigma^2 dt \\
 \geq E(\int_{0}^{T} f(\pi(t)) dt)
\end{align}
Dann muss es aber ein $t$ geben mit
\begin{equation}
 f(E(\pi(t))) > f(\pidach(t))
\end{equation}
Da $\pidach \leq \pistar$ und $f$ bis $\pistar$ streng monoton wachsend ist folgt
\begin{equation} \label{forsomet}
 E(\pi(t)) \geq \pidach(t).
\end{equation}

Wir bezeichen mit $\vdach$ den erwarteten logaritmischen Nutzen durch Verfolgung der Handelsstrategie
$\pidach$. Wir geben sie hier nochmal explizit an und berechnen dir zur Anwendung der Ito-Formel
benötigten partiellen Ableitungen:
\begin{align}
\vdach(t,x) &= \log(x)  +  r(T-t) + \int_{t}^{T}  \pidach(u)(\mu -r)-\frac{1}{2}\pidach(u)^2\sigma^2du \\
\vdach_x(t,x) &= \frac{1}{x} \\
\vdach_{xx}(t,x) &= - \frac{1}{x^2} \\
\vdach_t(t,x) &= -\pidach(t)(\mu -r)+\frac{1}{2}\pidach(t)^2\sigma^2 -r
\end{align}
Wir wiedeholen hier auch nochmal die $X$ treibende Dynamik um im folgenden die Anwendung der Ito-Formel
nachvollziehen zu können:
\begin{align}
 dX(t) &= X(t)(r + \pi(t)(\mu -r)) dt +  X(t) \sigma dW(t) \\
  X(0) &= x_0
\end{align}
Nun machen wir folgende Rechnung
\begin{align}
v_0(t,(1-\pi(t)\kstar)) & = E(\log(X'^{\pi})) \\
 &= \vdach(t,X(t)) \\
 &= \vdach(0,x) \\
 &+ \int_0^t \vdach_x(s,X) \pidach(s) \sigma dW(s) \\
 &+ \int_0^t \vdach_t(s,X) +  \vdach_x(s,X)(r+\pidach(s)(\mu-r)) + \einhalb \vdach_{xx}(s,X)\pidach(s)^2
\sigma^2 ds
\end{align}
Nehmen wir nun davon den Erwartungswert, so fällt das Integral bezüglich der Brownschen Bewegung dank
unserser Vorrausetzungen an eine zulässige Handelsstartegie weg, für das andere Integral berechnen wir:
\begin{align}
&=  \int_0^t -\pidach(t)(\mu -r)+\frac{1}{2}\pidach(t)^2\sigma^2 -r + (r+\pidach(s)(\mu-r)) +
\einhalb \pidach(s)^2 \sigma^2 ds \\
 & =  \int_0^t E(-\pi(t)(\mu -r)+\frac{1}{2}\pi(t)^2\sigma^2) -r + (r+\pidach(s)(\mu-r)) +
\einhalb \pidach(s)^2 \sigma^2 ds \\
 &= 0
\end{align}
Dabei haben wir die Ableitungen eingesetzt, Integral und Erwartungswert vertauscht.
Wir erhalten also
\begin{align}
E(v_0(t,\xdach (t) (1-\pi(t)\kstar))) & = E (\vdach(t,X(t))) \\
 &= \vdach(0,x) \\
 &= \vdach(T,X(T))
\end{align}
Also gilt:
\begin{align} \label{worstcaseboundpidach}
 WC(\pidach) &=  \min(\underset{t}{ \text{inf}} \quad E(v_0(t,\xdach (t) (1-\pi(t)\kstar))) , \vdach(0,x)) \\ 
             &= \min(\underset{t}{ \text{inf}} \quad \vdach(0,x) , \vdach(0,x)) \\
             &= \vdach(0,x)
\end{align}
Fall $\pi(0) \geq \pidach(0)$:  Dann gilt:
\begin{align}
 WC(\pi) &\leq  E(v_0(0,X (1-\pi(0)\kstar))) \\
          &=  v_0(0,x (1-\pi(0)\kstar) \\
          &\leq  v_0(0,x (1-\pidach(0)\kstar) \\
          &= \vdach(0,x) \\
          &= WC(\pidach)
\end{align}
Fall $\pi(0) \leq \pidach(0)$:
Wir betrachten dann
\begin{equation}
 \tau = \inf(t;E(\pi(t))>\pidach(t)).
\end{equation}
Wegen \ref{forsomet} liegt dieses $\tau$ zwischen 0 und $T$.

%habe ich noch nicht verstanden
Offensichtlich gilt dann auch:
\begin{equation} \label{damdam}
 E( \log (X^{\pi}(\tau)))  \leq E(\log(\xdach(\tau)))
\end{equation}

\begin{align} \label{dumdum}
 E(\log(1-\pi(t)\kstar)) &\leq \log(1-\kstar E(\pi(t)) \qquad \text{Jensens Ungleichung} \\
                       &< \log(1-\kstar \pidach(t)) \qquad \text{wegen $\ref{forsomet}$} 
\end{align}
 
Nun folgern wir
\begin{align}
 WC(\pi) &\leq  E(v_0(t,X (1-\pi(t)\kstar))) \\
          &= E(\log(X^{\pi}(t))) + E(1-\pi(t)\kstar) + ... \\ % explizite Form von v0
          &< E(\log(X^{\pi}(t))) + \log(1-\kstar \pidach(t)) + .. \qquad \text{wegen $\ref{dumdum}$} \\ 
          &< E(\log(\xdach(\tau))) + \log(1-\kstar \pidach(t)) + .. \qquad \text{wegen $\ref{damdam}$} \\
          &= E(v_0(t,\xdach(\tau)(1-\pi(t)\kstar))) \\
          &= WC(\pidach) 
\end{align}

Es bleibt noch der Fall, wo $\pi$ einen kleineren erwarteten Nutzen im crashfreien Scenario besitzt. Dann
gilt:
\begin{align}
 WC(\pi) &\leq  E(\log(X^{\pi}(T)))  \qquad \text{Definition von WC} \\
          &= E(\log(\xdach(T)))  \qquad \text{nach Vorrausetzung in diesem Fall} \\ % 
          &\leq WC(\pidach) \qquad \text{ wegen \ref{worstcaseboundpidach}} \\ 
\end{align}



\end{proof}


Wir haben noch nicht gezeigt, dass $\pidach$ wirklich eine bessere Worst-Case Schranke besitzt als die
Bondstrategie. Dass tun wir nun, indem wir eine Stategie angeben, die eine höhere Worstcaseschranke als die
pure Bondstrategie hat. Da $\pidach$ optimal ist, muss es auch eine bessere Worstcaseschranke besitzen. Für
eine numerische Berechnung verweisen wir auf das letzte Kapitel.

\begin{mySatz}
 Es gibt eine Strategie, mit einer besseren Worstcaseschranke.
\end{mySatz}

\begin{proof}
\newcommand{\besser}{\frac{1}{\kstar}(1-\exp(-\einhalb \frac{\pi-\mu}{\sigma}^2(T-t)))}
 Wir betrachten die Strategie
\begin{equation}
 \pi(t)=\einhalb \min(\besser,\pistern)
\end{equation}
Also gilt:
\begin{equation}
 0 < \pi(t) \leq \pistern
\end{equation}
Also folgt $f(\pi(t))>0$ und auch
\begin{equation} \label{besserohnecrash}
 0 < E \left[ 
	      \int_t^T f(\pi(s)) ds
       \right]
\end{equation}
Weiter gilt:

\begin{align*}
 v_0(t,x(1-\pi(t))\kstar)
 &= \log(x) + \log((1-\pi(t))\kstar) + \letzterTermv \\
 &> \log(x) + \log(1-(1-\exp(-\einhalb \frac{\pi-\mu}{\sigma}^2(T-t)))) + \letzterTermv \\
 &= \log(x) + -\einhalb \frac{\pi-\mu}{\sigma}^2(T-t) + \letzterTermv \\
 &= \log(x) + r(T-t) \\
 &= \log(x \exp(r(T-t))) 
\end{align*}
Dabei ergibt sich die Ungleichung, da $\log(1-x\kstar)$ streng fallend in $x$ ist und
\begin{equation*}
 \pi(t) < 2 \pi(t) \leq \besser
\end{equation*}
Also gilt:
\begin{equation} \label{besserbeicrash}
 v_0(t,x(1-\pi(t))\kstar)
 = \log(x \exp(r(T-t))) 
\end{equation}
Also folgt aus \ref{besserbeicrash} und \ref{besserohnecrash} das $\pi$ eine bessere Strategie als die reine
Bonddstrategie ist.
\end{proof}


Wir wollen nun noch den besten konstanten Portfoliprozess ausrechen:
\begin{mySatz}
 Der beste konstante Portfolioprozess ist:
\begin{equation}
 \pi= 
 \left(
  \einhalb  \left( \pistern + \frac{1}{\kstar} \right) - \sqrt{ 
   \frac{1}{4}  \left( \pistern - \frac{1}{\kstar} \right)^2 + \frac{1}{\sigma^2T}
   } 
\right)^+
\end{equation}
Weiter gilt: $\pi -> \min(\pistar,\frac{1}{\kstar})$ für $T->\infty$.
\end{mySatz}

\begin{proof}
 Zunächst zeigen wir die Grenzwertaussage. Geht T gegen $\infty$ so nimmt $\pi$ folgende Gestalt an:
\begin{equation}
 \pi= 
 \left(
  \einhalb  \left( \pistern + \frac{1}{\kstar} \right) - 
   \einhalb  | \pistern - \frac{1}{\kstar} |
\right)^+
\end{equation}
Daraus wird die Aussage offensichtlich.

Nun zur eigentlichen Aussage: Wir geben zunächst die Worst-Case-Schranke für einen konstanten
Portfolioprozess an:
\begin{equation}
 WC() = \log(x) + rT + \left( \pi(\mu-r)-\einhalb \pi^2 \sigma^2 \right) T + \log(1-\pi \kstar)
\end{equation}
Nun suchen wir dasjenige $\pi$ das diesen Ausdruck maximiert. Ableiten nach $\pi$ ergibt:
\begin{equation*}
  \left( (\mu-r)- \pi \sigma^2 \right) T + \frac{-\kstar}{1-\pi \kstar}
\end{equation*}
Gleichsetzen mit 0 und Multiplikation mit $1-\pi \kstar$ ergibt
\begin{equation*}
   (\mu-r)T - (\mu-r) T \pi \kstar - \pi \sigma^2 T + \pi^2 \sigma^2 T \kstar - \kstar = 0
\end{equation*}

Nun sortieren wir anch den Koeffizienten von $\pi$.
\begin{equation*}
  \pi \sigma^2 T -\pi T (( \mu-r) \kstar + \sigma^2) + (\mu-r)T-\kstar = 0
\end{equation*}

Und normieren noch:
\begin{equation*}
  \pi^2  -\pi T (\pistar + \frac{1}{\kstar}) + \frac{(\mu-r)}{\sigma^2 \kstar} - \frac{1}{\sigma^2T} = 0
\end{equation*}





\end{proof}
